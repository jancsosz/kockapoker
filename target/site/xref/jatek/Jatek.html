149     /** Kör nyertesét meghatározó függvény.
150      *  Először előállításra kerül egy ArrayList, amiben játékosonként, további ArrayList-ekben
151      *  el van tárolva, hogy az adott számból hányat dobott a körben.
152      *  Ezt az ArrayList-et elemenként vizsgálja az összes lehetséges sorozatra,
153      *  amiből végül játékosonként egyetles érték áll elő.
154      *  Ezeket az értékeket egy újabb ArrayListben tárolja el és ezek közül maximum-kiválasztással
155      *  meghatározza a legnagyobbat. Ha több teljesen megegyező van, akkor aki először dobta
156      *  az eredményt az nyer.
157      *
158      * @param korDobasok a játékosok adott körben dobott értékeinek listái egy listában összefoglalva.
159      * @return a nyertes játékos indexe.
160      */
161     private int korNyertes(ArrayList<ArrayList<Integer>> korDobasok){
162         ArrayList<Integer> mibolMennyi = new ArrayList<>();
163         ArrayList<ArrayList<Integer>> mibolMennyiLista = new ArrayList<>();
164         int darab = 0;
165         for (int i = 0; i < this.jatekosok.size(); ++i){
166             for (int j = 1; j <= 6; ++j){
167                 for (int k = 0; k < 5; ++k){
168                     if (korDobasok.get(i).get(k) == j){
169                         ++darab;
170                     }
171                 }
172                 mibolMennyi.add(darab);
173                 darab = 0;
174             }
175             mibolMennyiLista.add(new ArrayList<>(mibolMennyi));
176             mibolMennyi.clear();
177         }
178 
179         ArrayList<Integer> eredmenyek = new ArrayList<>();
180         for (int i = 0; i < this.jatekosok.size(); ++i){
181             if (par(mibolMennyiLista.get(i)) != 0) eredmenyek.add(par(mibolMennyiLista.get(i)));
182             else if (terc(mibolMennyiLista.get(i)) != 0) eredmenyek.add(terc(mibolMennyiLista.get(i)));
183             else if (ketPar(mibolMennyiLista.get(i)) != 0) eredmenyek.add(ketPar(mibolMennyiLista.get(i)));
184             else if (kisSor(mibolMennyiLista.get(i)) != 0) eredmenyek.add(kisSor(mibolMennyiLista.get(i)));
185             else if (nagySor(mibolMennyiLista.get(i)) != 0) eredmenyek.add(nagySor(mibolMennyiLista.get(i)));
186             else if (full(mibolMennyiLista.get(i)) != 0) eredmenyek.add(full(mibolMennyiLista.get(i)));
187             else if (kisPoker(mibolMennyiLista.get(i)) != 0) eredmenyek.add(kisPoker(mibolMennyiLista.get(i)));
188             else if (nagyPoker(mibolMennyiLista.get(i)) != 0) eredmenyek.add(nagyPoker(mibolMennyiLista.get(i)));
189             else eredmenyek.add(0);
190         }
191 
192         int nyertesIndex = 0;
193         int i = 0;
194         while (nyertesIndex == 0 && i < this.jatekosok.size()) {
195             if (eredmenyek.get(i).equals(Collections.max(eredmenyek))) nyertesIndex = i;
196             i++;
197         }
198 
199         return nyertesIndex;
200     }

206     private void kieses(){

209 
210             if (this.jatekosok.get(min).getNyertJatekokSzama() > this.jatekosok.get(i).getNyertJatekokSzama()){
211                 min = i;
212             }
213         }
214 
215         int leggyengebb = 0;
216         for (Jatekos jatekos : this.jatekosok) {
217             if (jatekos.getNyertJatekokSzama() == this.jatekosok.get(min).getNyertJatekokSzama()) {
218                 ++leggyengebb;
219             }
220         }
221 
222 
223         if (leggyengebb == 1) {
224             System.out.println("\n" + this.jatekosok.get(min).getNev() + " kiesett.");
225             this.jatekosok.remove(min);
226         }
227     }

257      * @param egyDobas egy játékos egy körben dobott értékei
258      * @return egy pár meghatározott értéke (100) + a pár rangja (párok közti megkülönböztetés céljából)
259      */
260     private int par(ArrayList<Integer> egyDobas){
261         int parSzam = 0;
262         int index = 0;
263         int tercSzam = 0;
264         for (int i = 0; i < 6; ++i){
265             if (egyDobas.get(i) == 2) {
266                 ++parSzam;
267                 index = i;
268             } else if (egyDobas.get(i) == 3) ++tercSzam;
269         }
270 
271         return parSzam == 1 && tercSzam == 0 ? (index + 1 + 100) : 0;
272     }
